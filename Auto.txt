    task.wait(0.1)  -- חשוב! מחכה שהמחיר והשם ייטענו

    local PriceFrame = child:FindFirstChild("Price")
    if not PriceFrame then return end

    local PriceLabel = PriceFrame:FindFirstChild("Amount")
    if not PriceLabel then return end

    local NameLabel = child:FindFirstChild("ITEM_NAME")
    if not NameLabel then return end

    local petName = NameLabel.Text
    local priceText = PriceLabel.Text

    -- ניקוי המחיר (למשל אם יש סימנים כמו , או רווחים)
    local cleanPrice = priceText:gsub("[^%d]", "")  -- מסיר הכל חוץ מספרים
    local currentPrice = tonumber(cleanPrice)

    if not table.find(desiredpet, petName) then return end
    if not currentPrice then return end

    local maxPrice = itemslists12[petName]
    if not maxPrice then return end

    if currentPrice <= maxPrice then
