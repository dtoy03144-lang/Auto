local function SendJobIdToFirebase(jobId)
    if not jobId or typeof(jobId) ~= "string" then
        warn("[Firebase] Invalid Job ID")
        return false
    end

    local data = {
        JobId = jobId,
        Timestamp = os.time(),
        IsoTime = DateTime.now():ToIsoDate(),
        Player = game.Players.LocalPlayer.Name,
        PlaceId = game.PlaceId,
        LastUpdated = tick()   -- זמן מדויק יותר
    }

    local success, response = pcall(function()
        return request({
            Url = FIREBASE_URL .. "tradebooth_servers/" .. jobId .. ".json" .. (FIREBASE_SECRET ~= "" and "?auth=" .. FIREBASE_SECRET or ""),
            Method = "PUT",        -- PUT = OVERWRITE מלא (מה שרצית)
            Headers = {
                ["Content-Type"] = "application/json"
            },
            Body = HttpService:JSONEncode(data)
        })
    end)

    if success then
        print("✅ Job ID Overwritten successfully:", jobId)
        return true
    else
        warn("❌ Failed to overwrite Job ID:", jobId, "| Error:", response)
        return false
    end
end
